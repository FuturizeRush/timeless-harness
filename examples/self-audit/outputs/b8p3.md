```html
<!doctype html>
<html lang="en">
<head>
  <meta charset="utf-8">
  <meta name="viewport" content="width=device-width, initial-scale=1">
  <title>Settings</title>
  <style>
    * { box-sizing: border-box; }
    body { margin: 0; font: 1rem/1.5 system-ui, sans-serif; color: #1b1b1b; }
    .panel { width: min(720px, calc(100% - 2rem)); margin: 2.5rem auto; }
    form { display: grid; gap: 1rem; }
    .field { display: grid; gap: .35rem; }
    input:not([type="checkbox"]), select, button { min-height: 44px; padding: .65rem; font: inherit; }
    .choice { display: flex; align-items: center; gap: .5rem; min-height: 44px; }
    button { border: 0; border-radius: .25rem; background: #1459d9; color: white; cursor: pointer; }
    button:disabled { cursor: wait; opacity: .7; }
    :focus-visible { outline: 3px solid #111; outline-offset: 2px; }
    #status { min-height: 1.5em; margin: 0; font-weight: 600; }
    .hint { margin: 0; color: #444; font-size: .9rem; }
  </style>
</head>
<body>
  <main class="panel">
    <h1>Settings</h1>
    <form id="settings-form">
      <div class="field">
        <label for="name">Display name</label>
        <input id="name" name="name" required maxlength="80" autocomplete="nickname"
               pattern=".*\S.*" title="Enter at least one non-space character."
               aria-describedby="name-hint">
        <p class="hint" id="name-hint">Enter 1–80 characters; spaces alone are not valid.</p>
      </div>

      <div class="field">
        <label for="timezone">Timezone</label>
        <select id="timezone" name="timezone" required>
          <option value="UTC">UTC</option>
          <option value="Asia/Taipei">Taipei</option>
        </select>
      </div>

      <label class="choice"><input id="email" name="email" type="checkbox"> Email me notifications</label>
      <button id="save" type="submit">Save settings</button>
      <p id="status" role="status" aria-live="polite" aria-atomic="true"></p>
    </form>
  </main>

  <script>
    const form = document.querySelector('#settings-form');
    const button = document.querySelector('#save');
    const status = document.querySelector('#status');

    form.addEventListener('submit', async (event) => {
      event.preventDefault();
      if (!form.reportValidity()) return;

      button.disabled = true;
      button.setAttribute('aria-busy', 'true');
      status.textContent = 'Saving settings…';

      try {
        const response = await fetch('/api/settings', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            name: form.elements.name.value.trim(),
            timezone: form.elements.timezone.value,
            email: form.elements.email.checked
          })
        });

        if (!response.ok) {
          const body = await response.json().catch(() => ({}));
          throw new Error(body.message || 'The server could not save your settings.');
        }
        status.textContent = 'Settings saved.';
      } catch (error) {
        status.textContent = `Save failed: ${error.message || 'Please try again.'}`;
      } finally {
        button.disabled = false;
        button.removeAttribute('aria-busy');
      }
    });
  </script>
</body>
</html>
```
