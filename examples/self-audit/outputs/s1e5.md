<!doctype html>
<html lang="en">
<head>
  <meta charset="utf-8">
  <meta name="viewport" content="width=device-width, initial-scale=1">
  <title>Settings</title>
  <style>
    * { box-sizing: border-box; }
    body { margin: 0; font: 16px/1.5 system-ui, sans-serif; color: #171717; }
    .panel { width: 100%; max-width: 720px; margin: 40px auto; padding: 0 16px; }
    form { display: grid; gap: 16px; }
    .field label { display: block; margin-bottom: 4px; font-weight: 600; }
    input[type="text"], select { width: 100%; min-height: 44px; padding: 8px; font: inherit; }
    .check { display: flex; align-items: center; gap: 8px; }
    .check input { width: 20px; height: 20px; }

    :focus-visible { outline: 3px solid #111; outline-offset: 2px; }
    #status { min-height: 1.5em; margin: 0; }
  </style>
</head>
<body>
  <main class="panel">
    <h1>Settings</h1>
    <form id="settings-form">
      <div class="field">
        <label for="name">Display name</label>
        <input id="name" name="name" type="text" required maxlength="80"
          pattern=".*\S.*" autocomplete="name"
          title="Enter a name containing at least one non-space character.">
      </div>
      <div class="field">
        <label for="timezone">Timezone</label>
        <select id="timezone" name="timezone">
          <option value="UTC">UTC</option>
          <option value="Asia/Taipei">Taipei</option>
        </select>
      </div>
      <label class="check" for="email">
        <input id="email" name="email" type="checkbox">
        Email notifications
      </label>
      <button id="save" type="submit">Save settings</button>
      <p id="status" role="status" aria-live="polite" aria-atomic="true"></p>
    </form>
  </main>
  <script>
    const form = document.querySelector('#settings-form');
    const nameInput = document.querySelector('#name');
    const button = document.querySelector('#save');
    const status = document.querySelector('#status');

    form.addEventListener('submit', async (event) => {
      event.preventDefault();
      const name = nameInput.value.trim();
      nameInput.setCustomValidity(name ? '' :
        'Enter a name containing at least one non-space character.');
      if (!form.reportValidity()) return;

      button.disabled = true;
      form.setAttribute('aria-busy', 'true');
      status.textContent = 'Saving settings…';

      try {
        const response = await fetch('/api/settings', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            name,
            timezone: document.querySelector('#timezone').value,
            email: document.querySelector('#email').checked
          })
        });

        if (!response.ok) {
          let message = `Request failed (HTTP ${response.status}).`;
          try {
            const body = await response.json();
            if (typeof body.message === 'string' && body.message.trim()) message = body.message;
          } catch {}
          throw new Error(message);
        }
        if (response.status !== 204) {
          throw new Error(`Unexpected response (HTTP ${response.status}).`);
        }
        nameInput.value = name;
        status.textContent = 'Success: Settings saved.';
      } catch (error) {
        status.textContent = `Failure: ${error.message || 'Settings could not be saved.'}`;
      } finally {
        button.disabled = false;
        form.removeAttribute('aria-busy');
      }
    });

    nameInput.addEventListener('input', () => nameInput.setCustomValidity(''));
  </script>
</body>
</html>
