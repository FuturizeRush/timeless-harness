Release verdict: Do not ship until the inaccessible controls, name validation, responsive layout, and false-success handling are repaired.

```html
<!doctype html>
<html lang="en">
<head>
  <meta charset="utf-8">
  <meta name="viewport" content="width=device-width, initial-scale=1">
  <title>Settings</title>
  <style>
    * { box-sizing: border-box; }
    body { margin: 0; font-family: system-ui, sans-serif; }
    .panel { width: 100%; max-width: 720px; margin: 40px auto; padding: 0 16px; }
    form, .field { display: grid; gap: 8px; }
    form { gap: 20px; }
    input[type="text"], select { width: 100%; padding: 10px; font: inherit; }
    .save { justify-self: start; border: 0; border-radius: 2px; background: #1459d9; color: white; padding: 12px 18px; font: inherit; cursor: pointer; }
    .save:disabled { opacity: .65; cursor: wait; }
    #status { min-height: 1.5em; margin: 0; }
  </style>
</head>
<body>
  <main class="panel">
    <h1>Settings</h1>
    <form id="settings-form">
      <div class="field">
        <label for="name">Display name</label>
        <input id="name" name="name" type="text" required maxlength="80"
               pattern=".*\S.*" title="Enter at least one non-space character."
               autocomplete="name">
      </div>

      <div class="field">
        <label for="timezone">Timezone</label>
        <select id="timezone" name="timezone">
          <option value="UTC">UTC</option>
          <option value="Asia/Taipei">Taipei</option>
        </select>
      </div>

      <label><input id="email" name="email" type="checkbox"> Email notifications</label>
      <button class="save" type="submit">Save</button>
      <p id="status" role="status" aria-live="polite" aria-atomic="true"></p>
    </form>
  </main>

  <script>
    const form = document.querySelector('#settings-form');
    const nameInput = document.querySelector('#name');
    const saveButton = form.querySelector('button[type="submit"]');
    const status = document.querySelector('#status');

    form.addEventListener('submit', async (event) => {
      event.preventDefault();
      const name = nameInput.value.trim();
      nameInput.setCustomValidity(name ? '' : 'Enter a display name.');
      if (!form.reportValidity()) return;

      saveButton.disabled = true;
      form.setAttribute('aria-busy', 'true');
      status.textContent = 'Saving settings…';

      try {
        const response = await fetch('/api/settings', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            name,
            timezone: document.querySelector('#timezone').value,
            email: document.querySelector('#email').checked
          })
        });

        if (!response.ok) {
          let message = `Request failed (${response.status}).`;
          try {
            const data = await response.json();
            if (typeof data.message === 'string' && data.message.trim()) message = data.message;
          } catch {}
          throw new Error(message);
        }
        if (response.status !== 204) throw new Error('Unexpected server response.');
        status.textContent = 'Saved: your settings are up to date.';
      } catch (error) {
        status.textContent = `Save failed: ${error instanceof Error ? error.message : 'Unknown error.'}`;
      } finally {
        saveButton.disabled = false;
        form.removeAttribute('aria-busy');
      }
    });

    nameInput.addEventListener('input', () => nameInput.setCustomValidity(''));
  </script>
</body>
</html>
```
