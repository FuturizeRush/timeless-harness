Release verdict: Do not ship until the responsive layout, validation, accessibility, and truthful request states are repaired.

```html
<!doctype html>
<html lang="en">
<head>
  <meta charset="utf-8">
  <meta name="viewport" content="width=device-width, initial-scale=1">
  <title>Settings</title>
  <style>
    * { box-sizing: border-box; }
    body { margin: 0; font: 16px/1.5 system-ui, sans-serif; }
    .panel { max-width: 720px; margin: 40px auto; padding: 0 16px; }
    #name, select { display: block; width: 100%; min-height: 44px; margin: 6px 0 16px; padding: 8px; font: inherit; }
    .preference { display: block; margin: 0 0 16px; }
    .save { border: 0; background: #1459d9; color: white; padding: 12px 18px; font: inherit; cursor: pointer; }
    .save:disabled { opacity: .65; cursor: wait; }
    :focus-visible { outline: 3px solid #111; outline-offset: 3px; }
    #status { min-height: 1.5em; }
  </style>
</head>
<body>
  <main class="panel">
    <h1>Settings</h1>
    <form id="settings-form">
      <label for="name">Display name</label>
      <input id="name" type="text" autocomplete="name" required maxlength="80">

      <label for="timezone">Timezone</label>
      <select id="timezone" required>
        <option value="UTC">UTC</option>
        <option value="Asia/Taipei">Taipei</option>
      </select>

      <label class="preference" for="email">
        <input id="email" type="checkbox"> Email notifications
      </label>

      <button class="save" type="submit">Save</button>
      <p id="status" role="status" aria-live="polite" aria-atomic="true"></p>
    </form>
  </main>

  <script>
    const form = document.querySelector('#settings-form');
    const nameInput = document.querySelector('#name');
    const button = document.querySelector('.save');
    const status = document.querySelector('#status');

    nameInput.addEventListener('input', () => nameInput.setCustomValidity(''));
    form.addEventListener('invalid', () => {
      status.textContent = 'Not saved: fix the highlighted field.';
    }, true);

    form.addEventListener('submit', async (event) => {
      event.preventDefault();
      const name = nameInput.value.trim();

      if (!name || name.length > 80) {
        nameInput.setCustomValidity('Enter a display name between 1 and 80 characters.');
        nameInput.reportValidity();
        status.textContent = 'Not saved: enter a valid display name.';
        return;
      }

      nameInput.value = name;
      button.disabled = true;
      form.setAttribute('aria-busy', 'true');
      status.textContent = 'Saving…';

      try {
        const response = await fetch('/api/settings', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            name,
            timezone: document.querySelector('#timezone').value,
            email: document.querySelector('#email').checked
          })
        });

        if (!response.ok) {
          let message = `Request failed (${response.status}).`;
          try {
            const data = await response.json();
            if (typeof data.message === 'string' && data.message.trim()) {
              message = data.message.trim();
            }
          } catch {}
          throw new Error(message);
        }

        status.textContent = 'Saved successfully.';
      } catch (error) {
        const message = error instanceof Error ? error.message : 'Unable to reach the server.';
        status.textContent = `Save failed: ${message}`;
      } finally {
        button.disabled = false;
        form.removeAttribute('aria-busy');
      }
    });
  </script>
</body>
</html>
```
