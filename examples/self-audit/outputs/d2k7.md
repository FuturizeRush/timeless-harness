Release verdict: Do not ship the current page; it lacks validation, accessible controls, responsive layout, and truthful failure handling.

<!doctype html>
<html lang="en">
<head>
  <meta charset="utf-8">
  <meta name="viewport" content="width=device-width, initial-scale=1">
  <title>Settings</title>
  <style>
    * { box-sizing: border-box; }
    body { margin: 0; color: #1b1b1b; font: 16px/1.5 system-ui, sans-serif; }
    .panel { width: min(100%, 720px); margin: 40px auto; padding: 0 16px; }

<body>
  <main class="panel">
    <h1>Settings</h1>
    <form id="settingsForm">
      <div class="field">
        <label for="displayName">Display name</label>
        <input id="displayName" name="name" type="text" required maxlength="80"
               autocomplete="name" aria-describedby="nameHint">
        <p class="hint" id="nameHint">1–80 characters; spaces alone are not valid.</p>
      </div>

      <div class="field">
        <label for="timezone">Timezone</label>
        <select id="timezone" name="timezone" required>
          <option value="UTC">UTC</option>
          <option value="Asia/Taipei">Taipei</option>
        </select>
      </div>

      <div class="field">
        <label class="check" for="email">
          <input id="email" name="email" type="checkbox">
          Email notifications
        </label>
      </div>

      <button id="saveButton" type="submit">Save settings</button>
      <p id="status" role="status" aria-live="polite" aria-atomic="true"></p>
    </form>
  </main>

  <script>
    const form = document.querySelector('#settingsForm');
    const nameInput = document.querySelector('#displayName');
    const button = document.querySelector('#saveButton');
    const status = document.querySelector('#status');
    const controls = form.querySelectorAll('input, select, button');
    let saving = false;

    form.addEventListener('input', () => {
      nameInput.setCustomValidity('');
      if (!saving) status.textContent = '';
    });

    form.addEventListener('submit', async (event) => {
      event.preventDefault();
      const name = nameInput.value.trim();
      nameInput.setCustomValidity(
        name ? '' : 'Enter a display name that is not only spaces.'
      );
      if (!form.reportValidity()) return;

      const payload = {
        name,
        timezone: document.querySelector('#timezone').value,
        email: document.querySelector('#email').checked
      };

      saving = true;
      controls.forEach((control) => { control.disabled = true; });
      form.setAttribute('aria-busy', 'true');
      button.textContent = 'Saving…';
      status.textContent = 'Saving settings…';

      try {
        let response;
        try {
          response = await fetch('/api/settings', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(payload)
          });
        } catch {
          throw new Error('Network error; your settings were not saved.');
        }

        if (!response.ok) {
          let message = `Server error (${response.status}).`;
          try {
            const data = await response.json();
            if (typeof data.message === 'string' && data.message.trim()) {
              message = data.message.trim();
            }
          } catch {}
          throw new Error(message);
        }
        if (response.status !== 204) {
          throw new Error(`Unexpected response (${response.status}); save was not confirmed.`);
        }
        status.textContent = 'Settings saved successfully.';
      } catch (error) {
        status.textContent = `Save failed: ${error.message}`;
      } finally {
        saving = false;
        controls.forEach((control) => { control.disabled = false; });
        form.removeAttribute('aria-busy');
        button.textContent = 'Save settings';
      }
    });
  </script>
</body>
</html>
