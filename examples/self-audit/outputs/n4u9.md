  <main class="panel">
    <h1>Settings</h1>
    <form id="settings-form" novalidate>
      <label for="name">Display name</label>
      <input id="name" name="name" type="text" autocomplete="name"
             required minlength="2" maxlength="80" aria-describedby="name-help">
      <p id="name-help" class="help">Use 2–80 characters, including at least one letter or number.</p>

      <label for="timezone">Timezone</label>
      <select id="timezone" name="timezone" required>
        <option value="UTC">UTC</option>
        <option value="Asia/Taipei">Taipei</option>
      </select>

      <label class="check" for="email">
        <input id="email" name="email" type="checkbox">
        Email notifications
      </label>

      <button id="save" type="submit">Save settings</button>
      <p id="status" class="status" role="status" aria-live="polite" aria-atomic="true"></p>
    </form>

  <script>
    const form = document.querySelector('#settings-form');
    const nameInput = document.querySelector('#name');
    const button = document.querySelector('#save');
    const status = document.querySelector('#status');

    nameInput.addEventListener('input', () => nameInput.setCustomValidity(''));

    form.addEventListener('submit', async (event) => {
      event.preventDefault();
      const name = nameInput.value.trim();
      const validName = name.length >= 2 && name.length <= 80 &&
        /[\p{L}\p{N}]/u.test(name) && !/[<>\p{Cc}]/u.test(name);
      nameInput.setCustomValidity(validName ? '' :
        'Enter 2–80 characters with at least one letter or number, and no angle brackets.');

      if (!form.checkValidity()) {
        status.textContent = 'Not saved: Fix the highlighted field.';
        form.reportValidity();
        return;
      }

      nameInput.value = name;
      button.disabled = true;
      form.setAttribute('aria-busy', 'true');
      status.textContent = 'Saving settings…';

      try {
        const response = await fetch('/api/settings', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            name,
            timezone: document.querySelector('#timezone').value,
            email: document.querySelector('#email').checked
          })
        });

        if (!response.ok) {
          let message = `Request failed (${response.status}).`;
          try {
            const data = await response.json();
            if (typeof data.message === 'string' && data.message.trim()) message = data.message.trim();
          } catch {}
          status.textContent = `Save failed: ${message}`;
          return;
        }
        status.textContent = 'Settings saved successfully.';
      } catch {
        status.textContent = 'Save failed: The server could not be reached. Try again.';
      } finally {
        button.disabled = false;
        form.removeAttribute('aria-busy');
      }
    });
  </script>
</body>
